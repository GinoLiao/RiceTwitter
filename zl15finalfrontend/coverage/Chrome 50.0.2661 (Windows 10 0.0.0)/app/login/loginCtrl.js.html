function LoginCtrl(apiService, UserService) {

    function login() {
		
		//*********************comment out  these two lines to pass jasmine test*********************//
		document.getElementById("mainLink").style.display = "inline-block"
		document.getElementById("profileLink").style.display = "inline-block"
		
		if (!vm.username) {
   	    console.log('username not defined') // this wasn't here before but is good to have!
    	    return
    	}
        console.log('logging in', vm.username) // you don't want comments like these in production!
         
        apiService.login({'username':vm.username, 'password':vm.password})
             .$promise.then(function(result) {
                   console.log('Payload from server:', result)
                   getStatus()
                   vm.loggedIn = true
                   vm.password = ''
                   // XXX grab the username from the server
                   // put it into the UserService singleton
				   vm.username = result.username
				   UserService.username = result.username
				   //*********************comment out  this line to pass jasmine test*********************//
				   window.location.href = window.location.origin + "/#/Main"
				   
             }, function(error){
				 vm.message = "User name and password don't match. Please try again"
			 })
		
    }
 
    function logout() {
		//*********************comment out  these two lines to pass jasmine test*********************//
		document.getElementById("mainLink").style.display = "none"
		document.getElementById("profileLink").style.display = "none"
		
		apiService.logout().$promise.then(function(result){
			vm.username = ''
			vm.userStatus =''
			vm.newUserStatus =''
			vm.loggedIn = false
			// XXX clear the UserService singleton's username value
			console.log('LogOut', result)
			getStatus()
			UserService.username = null //XXX
		})
		
    }
 
 
	function loadPosts() {
		vm.posts.length = 0
		apiService.getPosts().$promise.then(function(result) {
			result.posts.forEach(function(post) {
				vm.posts.push(post)
			})
		})
		console.log("in login ctrl  " + vm.posts.length)
		UserService.posts = vm.posts
	}
 
 
	function updateStatus() {
		apiService.setStatus({ status: vm.newUserStatus}).$promise.
		then(function(result) {
            console.log(result)
			vm.userStatus = result.status
			vm.newUserStatus = vm.userStatus
		})
	}
 
  
  function getStatus() {
    apiService.getStatus().$promise.then(function(result) {
      console.log('status', result.statuses[0])
      vm.userStatus = result.statuses[0].status
	  UserService.userStatus = vm.userStatus
	  vm.newUserStatus = vm.userStatus
      vm.message = 'logged in as ' + vm.username + ' with status "' + vm.userStatus + '"'
    }, function(error) {
      console.log('User not logged in', error)
      vm.message = 'Not Logged In'
    })
  }
}
